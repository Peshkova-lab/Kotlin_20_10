class Student(val id:Int, var name: String, var lastname: String, var group: Int, var facultet: String) {

    

    fun accept (students: MutableList<Student>) {

        

      var check: Boolean = true;

        

      for (student in students) {

          if (student.id == this.id) { check = false; }

      }

      

      if (check) {

        students.add(this)

        println ("Success!")

      }

      else println("This student exists!");

    }

    

    fun delete(students: MutableList<Student>) {

        var check: Boolean = false;

        

      for (student in students) {

          if (student.id == this.id) { check = true; }

      }

      

      if (check) {

          students.remove(this);

        println ("Success!");

      }

    }

      

}
